Module: mustache/s_render

Renders a mustache template given a supporting JSON object

Parameters:
Name Type Description
@template nvarchar(max)

the mustache template to render
@json nvarchar(max)

the JSON object that will be used to render the sections and tags within the sub-document
Author:
  • James Kelly
See:
  • https://mustache.github.io/
Returns:

the @template rendered according to the {{mustache}} specification

Type
nvarchar(max)
Example
DECLARE @tpl NVARCHAR(max) = N'<h1>{{header}}</h1>
{{#bug}}
{{/bug}}

<ul>
{{#items}}{{#first}}<li><strong>{{name}}</strong></li>{{/first}}{{#link}}<li><a href="{{url}}">{{name}}</a></li>{{/link}}
{{/items}}</ul>

{{#empty}}  <p>The list is empty.</p>{{/empty}}

<p>{{notset}}{{! comment}}</p>
';

DECLARE @obj NVARCHAR(max)	= N'{
  "header": "Colors",
  "items": [
      {"name": "red", "first": true, "url": "#Red"},
      {"name": "green", "link": true, "url": "#Green"},
      {"name": "blue", "link": true, "url": "#Blue"}
  ],
  "empty": true
}';

PRINT mustache.s_render(@tpl, @obj);
